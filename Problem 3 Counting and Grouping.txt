def item_order(order):
    words = order.split(" ")
    sal = 0
    wat = 0
    hum = 0
    for word in words:
        if word == 'salad':
            sal += 1
        elif word == 'hamburger':
            hum += 1
        elif word == 'water':
            wat += 1
            
    return "salad:" + str(sal) + " " + "hamburger:" + str(hum) + " " + "water:" + str(wat)
